# Test command script for the Monochron emulator
#
# Purpose: Test glcdFillRectangle2()
#

# Give help message
hm --- press 'q' to quit ---

# Erase lcd display and paint the function name
le
pa f 0 0 5x5p h 1 1 glcdFillRectangle2

# Enclose the paint area with a big rectangle
pr f 3 6 122 57

# Indefinitely keep animating filled rectangles of
# mixed length and width
wts
rf pattern=5 true pattern=(pattern==0?5:pattern-1)
  vs dx=1
  rf x=0 x<14 x=x+1
    vs dy=1
    rf y=0 y<9 y=y+1
      prf f x+dx+4 y+dy+7 x+1 y+1 2 (x+y+pattern)%6
      vs dy=y+dy+1
    rn
    vs dx=x+dx+1
  rn
  wte 100
rn
